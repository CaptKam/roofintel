5. Full Python Example (copy-paste ready)
Pythonimport requests
import geopandas as gpd
from shapely.geometry import Polygon

def get_hail_threats(lat, lon):
    url = f"https://data.api.xweather.com/hail/threats/closest"
    params = {
        "client_id": "YOUR_ID",
        "client_secret": "YOUR_SECRET",
        "loc": f"{lat},{lon}",
        "p": 20,          # limit results
        "format": "geojson"  # if available, otherwise parse manually
    }
    r = requests.get(url, params=params)
    return r.json()
