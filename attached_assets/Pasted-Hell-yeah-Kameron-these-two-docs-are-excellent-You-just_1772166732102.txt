2. The Agent — server/enrichment-roi-agent.ts (full file)
TypeScript// server/enrichment-roi-agent.ts
import { db } from './db';
import { leads, enrichmentDecisions, markets } from '../shared/schema';
import { eq, and, gte, inArray } from 'drizzle-orm';
import { recordEvidence } from './evidence-recorder'; // reuse your existing

// Config — move to env + admin UI later
const ROI_CONFIG = {
  MIN_ROI_THRESHOLD: 8.0,           // 8:1 minimum (tune per market)
  AVG_DEAL_SIZE: 28500,             // commercial roof job — make per-market
  BASE_CLOSE_RATE: 0.09,            // 9% historical — will become ML later
  HAIL_SEASON_MULTIPLIER: 1.8,      // Mar–Jun
  MAX_SPEND_PER_LEAD: 1.25,
  TIERS: {
    tier1: { cost: 0.18, apis: ['google_places_phone'] },           // cheap
    tier2: { cost: 0.45, apis: ['batchdata_phone', 'hunter_email'] },
    tier3: { cost: 0.95, apis: ['pdl_full', 'batchdata_skip'] },
    premium: { cost: 2.50, apis: ['lexisnexis_full'] },
  },
};

interface RoiDecision {
  decisionType: string;
  roiScore: number;
  expectedValue: number;
  cost: number;
  recommendedApis: string[];
  reason: string;
  confidence: number;
}

export async function processOneLead(lead: Lead, marketConfig?: any): Promise<RoiDecision> {
  const now = new Date();
  const isHailSeason = now.getMonth() >= 2 && now.getMonth() <= 5; // Mar–Jun

  // 1. Base signals from your existing data (all free)
  const baseScore = lead.leadScore || 0;
  const contactability = (lead.ownerPhone ? 1 : 0) + (lead.ownerEmail ? 1 : 0) + (lead.contactName ? 1 : 0);
  const missingHighValueFields = [
    !lead.ownerPhone,
    !lead.contactEmail,
    !lead.managingMemberPhone,
    lead.hailEvents < 3 && lead.roofLastReplaced == null,
  ].filter(Boolean).length;

  const upliftPotential = Math.min(0.45, missingHighValueFields * 0.12); // max 45% lift

  // 2. Market & seasonality factors
  const marketFactor = 1 + (lead.hailEvents / 20) + (lead.isFloodHighRisk ? 0.3 : 0);
  const seasonalMultiplier = isHailSeason ? ROI_CONFIG.HAIL_SEASON_MULTIPLIER : 1.0;

  // 3. Expected Value (EV)
  const contactMultiplier = contactability === 3 ? 1.0 : contactability === 2 ? 0.75 : 0.4;
  const ev = baseScore 
    * contactMultiplier 
    * marketFactor 
    * seasonalMultiplier 
    * (1 + upliftPotential) 
    * ROI_CONFIG.AVG_DEAL_SIZE 
    * ROI_CONFIG.BASE_CLOSE_RATE;

  // 4. Tiered cost & ROI
  let tier = 'skip';
  let cost = 0;
  let recommendedApis: string[] = [];
  let reason = '';

  if (baseScore < 45 || contactability === 3) {
    tier = 'free_only';
    reason = 'Already high contactability or low base score';
  } else if (ev < 800) {
    tier = 'skip';
    reason = 'EV too low for any paid spend';
  } else if (ev < 2500) {
    tier = 'tier1';
    ({ cost, apis: recommendedApis } = ROI_CONFIG.TIERS.tier1);
    reason = 'Low-medium EV — cheap phone append only';
  } else if (ev < 6500) {
    tier = 'tier2';
    ({ cost, apis: recommendedApis } = ROI_CONFIG.TIERS.tier2);
    reason = 'Strong roof/hail signals — phone + email';
  } else if (ev < 12000) {
    tier = 'tier3';
    ({ cost, apis: recommendedApis } = ROI_CONFIG.TIERS.tier3);
    reason = 'High-value lead — full skip trace';
  } else {
    tier = 'premium';
    ({ cost, apis: recommendedApis } = ROI_CONFIG.TIERS.premium);
    reason = 'Portfolio monster / critical hail exposure — premium full package';
  }

  const roiScore = cost > 0 ? ev / cost : 0;
  const finalDecision = roiScore >= ROI_CONFIG.MIN_ROI_THRESHOLD ? tier : 'skip';
  const confidence = Math.min(95, Math.round(60 + (baseScore / 2) + (contactability * 8)));

  // 5. Record decision as evidence
  await recordEvidence({
    leadId: lead.id,
    contactType: 'enrichment_decision',
    contactValue: finalDecision,
    sourceName: 'enrichment-roi-agent',
    sourceType: 'ALGORITHM',
    confidence,
  });

  // 6. Save to decisions table
  await db.insert(enrichmentDecisions).values({
    leadId: lead.id,
    marketId: lead.marketId,
    decisionType: finalDecision,
    roiScore,
    expectedValue: ev,
    enrichmentCost: cost,
    recommendedApis,
    confidence,
    reasonSummary: reason,
  });

  return {
    decisionType: finalDecision,
    roiScore,
    expectedValue: ev,
    cost,
    recommendedApis,
    reason,
    confidence,
  };
}

export async function runBatch(marketId?: string, filterLeadIds?: string[]) {
  let query = db.select().from(leads).where(
    and(
      eq(leads.enrichmentStatus, 'complete'), // only enriched leads
      marketId ? eq(leads.marketId, marketId) : undefined,
      filterLeadIds ? inArray(leads.id, filterLeadIds) : undefined,
      gte(leads.leadScore, 40) // quick filter
    )
  );

  const batchLeads = await query;
  const results: any[] = [];

  for (let i = 0; i < batchLeads.length; i += 100) { // safe chunking
    const chunk = batchLeads.slice(i, i + 100);
    await Promise.all(chunk.map(async (lead) => {
      try {
        const decision = await processOneLead(lead);
        results.push({ leadId: lead.id, ...decision });
      } catch (err) {
        console.error(`ROI failed for ${lead.id}`, err);
      }
    }));
    await new Promise(r => setTimeout(r, 300)); // gentle rate limit
  }

  return { processed: results.length, decisions: results };
}
