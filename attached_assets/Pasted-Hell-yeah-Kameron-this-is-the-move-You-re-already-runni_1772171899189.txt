2. Grok Proxy (server/domains/intelligence/grok-proxy.ts)
TypeScript// server/domains/intelligence/grok-proxy.ts
import { Grok } from '@xai/grok-sdk'; // official 2026 SDK (or openai-compatible)
import { db } from '../../db';
import { enrichmentBudgets } from '../../shared/schema';
import { eq } from 'drizzle-orm';

const grok = new Grok({
  apiKey: process.env.XAI_GROK_API_KEY!,
  model: 'grok-4.1-fast', // reasoning + tool calling
});

export async function callGrokCore(prompt: string, context: any, tools: any[] = []) {
  const marketConfig = await db.query.enrichmentBudgets.findFirst({
    where: eq(enrichmentBudgets.marketId, context.marketId || 'dfw'),
  });

  // System prompt — authentic Grok style
  const systemPrompt = `You are RoofIntel's Grok Intelligence Core — truthful, maximally helpful, hail-chasing focused, commercial roofing expert.
You have perfect knowledge of the domain: hail physics, insurance claim windows, LLC chains, ROI math, Texas counties.
Always reason step-by-step. Always cite evidence. Never hallucinate sources.
Current market: ${context.marketId}. Budget remaining today: $${marketConfig?.dailyBudgetUsd - (marketConfig?.spentTodayUsd || 0)}.
ROI gate is mandatory — never suggest enrichment without positive ROI.`;

  const response = await grok.chat.completions.create({
    messages: [
      { role: 'system', content: systemPrompt },
      ...context.history || [],
      { role: 'user', content: prompt },
    ],
    tools: tools.map(t => t.schema), // OpenAI tool format
    tool_choice: 'auto',
    temperature: 0.2,
    max_tokens: 2048,
  });

  // Log trace
  await db.insert(/* agent_traces table */).values({
    sessionId: context.sessionId,
    prompt,
    response: response.choices[0].message,
    tokensUsed: response.usage.total_tokens,
    costUsd: (response.usage.total_tokens / 1_000_000) * 0.35, // Fast tier
  });

  return response.choices[0].message;
}
3. Supervisor (server/domains/intelligence/supervisor.ts)
TypeScript// server/domains/intelligence/supervisor.ts
import { callGrokCore } from './grok-proxy';
import * as dbTool from './tools/db-query.tool';
import * as roiTool from './tools/roi-trigger.tool';
import * as zipTool from './tools/zip-compute.tool';
import * as webTool from './tools/web-browse.tool';
import * as xTool from './tools/x-search.tool';

const ALL_TOOLS = [dbTool, roiTool, zipTool, webTool, xTool];

export async function runGrokSupervisor(userPrompt: string, context: any) {
  let history = context.history || [];
  let maxSteps = 8; // safety

  while (maxSteps-- > 0) {
    const message = await callGrokCore(userPrompt, { ...context, history }, ALL_TOOLS);

    history.push({ role: 'assistant', content: message.content });

    if (!message.tool_calls?.length) {
      // Final answer
      return { plan: message.content, history, confidence: 85 };
    }

    // Execute tools (parallel safe)
    for (const toolCall of message.tool_calls) {
      const tool = ALL_TOOLS.find(t => t.name === toolCall.function.name);
      if (!tool) continue;

      const result = await tool.execute(JSON.parse(toolCall.function.arguments));
      history.push({
        role: 'tool',
        tool_call_id: toolCall.id,
        content: JSON.stringify(result),
      });
    }
  }

  return { plan: 'Max reasoning steps reached', history };
}
