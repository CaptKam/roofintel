T002: Enrichment ROI Agent (server/enrichment-roi-agent.ts)

TypeScript// server/enrichment-roi-agent.ts
import { db } from './db';
import { leads, enrichmentDecisions, enrichmentBudgets } from '../shared/schema';
import { eq, and, gte, inArray, desc, sql } from 'drizzle-orm';
import { recordEvidence } from './evidence-recorder';

let batchProgress = { processed: 0, total: 0 };

export async function getMarketConfig(marketId: string) {
  const config = await db.query.enrichmentBudgets.findFirst({ where: eq(enrichmentBudgets.marketId, marketId) });
  return config || {  // fallback defaults
    hailSeasonMultiplier: 1.8,
    minRoiThreshold: 8.0,
    avgDealSize: 28500,
    baseCloseRate: 0.09,
    dailyBudgetUsd: 500,
    spentTodayUsd: 0,
  };
}

export async function processOneLead(lead: Lead, marketConfig: any): Promise<any> {
  const now = new Date();
  const isHailSeason = now.getMonth() >= 2 && now.getMonth() <= 5;
  const daysSinceHail = lead.lastHailDate ? Math.floor((now.getTime() - new Date(lead.lastHailDate).getTime()) / 86400000) : 999;

  // Early exits
  if (lead.dncRegistered || ['government', 'reit', 'nonprofit'].includes(lead.ownershipStructure || '')) {
    return { decisionType: 'skip', roiScore: 0, expectedValue: 0, cost: 0, recommendedApis: [], reason: 'DNC or non-target owner', confidence: 95 };
  }
  const contactability = (lead.ownerPhone ? 1 : 0) + (lead.ownerEmail ? 1 : 0) + (lead.contactName ? 1 : 0) + (lead.managingMemberPhone ? 1 : 0);
  if (contactability >= 3 && lead.intelligenceScore > 92) {
    return { decisionType: 'free_only', roiScore: 0, expectedValue: 0, cost: 0, recommendedApis: [], reason: 'High intelligence + full contacts', confidence: 90 };
  }

  // Hail-chaser multipliers
  const hailFactor = Math.min(3.2, 
    (lead.hailEvents || 0) * 0.35 + 
    (lead.claimWindowOpen ? 2.2 : 0) + 
    (daysSinceHail < 14 ? 2.8 : 0)
  );

  const missingHighValue = [!lead.ownerPhone, !lead.contactEmail, !lead.managingMemberPhone, lead.hailEvents < 3 && !lead.roofLastReplaced].filter(Boolean).length;
  const uplift = Math.min(0.45, missingHighValue * 0.12);

  const contactMultiplier = contactability >= 3 ? 1.0 : contactability === 2 ? 0.75 : 0.4;
  const marketFactor = 1 + ((lead.hailEvents || 0) / 20) + (lead.isFloodHighRisk ? 0.3 : 0);
  const seasonal = isHailSeason ? marketConfig.hailSeasonMultiplier : 1.0;

  const ev = (lead.leadScore || 40) 
    * contactMultiplier 
    * marketFactor 
    * seasonal 
    * (1 + uplift) 
    * hailFactor 
    * marketConfig.avgDealSize 
    * marketConfig.baseCloseRate;

  // Tier logic
  let tier = 'skip', cost = 0, recommendedApis: string[] = [], reason = '';
  if (ev < 800) { tier = 'skip'; reason = 'EV too low'; }
  else if (ev < 2500) { tier = 'tier1'; cost = 0.18; recommendedApis = ['google_places_phone']; reason = 'Cheap phone append'; }
  else if (ev < 6500) { tier = 'tier2'; cost = 0.45; recommendedApis = ['batchdata_phone', 'hunter_email']; reason = 'Strong hail signals'; }
  else if (ev < 12000) { tier = 'tier3'; cost = 0.95; recommendedApis = ['pdl_full', 'batchdata_skip']; reason = 'High-value target'; }
  else { tier = 'premium'; cost = 2.50; recommendedApis = ['lexisnexis_full']; reason = 'Portfolio + fresh hail'; }

  const roiScore = cost > 0 ? ev / cost : 0;
  const finalDecision = (roiScore >= marketConfig.minRoiThreshold && cost <= 1.25) ? tier : 'skip';

  const confidence = Math.min(95, 55 + Math.round((lead.leadScore || 0) / 2) + contactability * 8 + hailFactor * 6);

  // Budget guardrail
  const dailyRemaining = marketConfig.dailyBudgetUsd - marketConfig.spentTodayUsd;
  if (finalDecision !== 'skip' && cost > dailyRemaining) {
    return { decisionType: 'skip', roiScore: 0, expectedValue: ev, cost: 0, recommendedApis: [], reason: 'Daily budget exhausted', confidence };
  }

  // Record
  await recordEvidence({ leadId: lead.id, contactType: 'enrichment_decision', contactValue: finalDecision, sourceName: 'roi-agent', sourceType: 'ALGORITHM', confidence });
  await db.insert(enrichmentDecisions).values({
    leadId: lead.id, marketId: lead.marketId, decisionType: finalDecision,
    roiScore, expectedValue: ev, enrichmentCost: cost, recommendedApis, confidence, reasonSummary: reason,
  });

  // Update spent (optimistic — reconcile nightly)
  if (cost > 0) {
    await db.update(enrichmentBudgets).set({ spentTodayUsd: sql`${enrichmentBudgets.spentTodayUsd} + ${cost}` })
      .where(eq(enrichmentBudgets.marketId, lead.marketId));
  }

  return { decisionType: finalDecision, roiScore, expectedValue: ev, cost, recommendedApis, reason, confidence };
}

export async function runBatch(marketId?: string, filterLeadIds?: string[], zipCode?: string) {
  batchProgress = { processed: 0, total: 0 };

  const marketConfig = await getMarketConfig(marketId || 'dfw');

  let where = and(gte(leads.leadScore, 40), eq(leads.enrichmentStatus, 'complete'));
  if (marketId) where = and(where, eq(leads.marketId, marketId));
  if (filterLeadIds) where = and(where, inArray(leads.id, filterLeadIds));
  if (zipCode) where = and(where, eq(leads.zipCode, zipCode));

  const batchLeads = await db.select().from(leads).where(where).orderBy(desc(leads.leadScore));
  batchProgress.total = batchLeads.length;

  const results: any[] = [];
  for (let i = 0; i < batchLeads.length; i += 100) {
    const chunk = batchLeads.slice(i, i + 100);
    await Promise.all(chunk.map(async lead => {
      const d = await processOneLead(lead, marketConfig);
      results.push({ leadId: lead.id, ...d });
      batchProgress.processed++;
    }));
    await new Promise(r => setTimeout(r, 300));
  }

  return { processed: results.length, decisions: results, totalEv: results.reduce((sum, d) => sum + d.expectedValue, 0) };
}

export function getBatchProgress() { return batchProgress; }

export async function getEnrichmentStats(marketId?: string) {
  const stats = await db.select({
    tier: enrichmentDecisions.decisionType,
    count: sql<number>`count(*)`,
    totalEv: sql<number>`sum(${enrichmentDecisions.expectedValue})`,
    totalCost: sql<number>`sum(${enrichmentDecisions.enrichmentCost})`,
  }).from(enrichmentDecisions)
    .where(marketId ? eq(enrichmentDecisions.marketId, marketId) : undefined)
    .groupBy(enrichmentDecisions.decisionType);

  const budget = await getMarketConfig(marketId || 'dfw');
  return { stats, budgetRemaining: budget.dailyBudgetUsd - budget.spentTodayUsd };
}

TypeScript// server/zip-tile-scoring.ts
import { db } from './db';
import { leads, zipTiles } from '../shared/schema';
import { eq, sql } from 'drizzle-orm';

export async function computeZipTile(zipCode: string, marketId: string) {
  const stats = await db.execute(sql`
    SELECT 
      COUNT(*)::int as lead_count,
      AVG(${leads.hailEvents}) as avg_hail,
      PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY ${leads.totalValue}) as median_value,
      AVG(CASE WHEN ${leads.ownerPhone} IS NULL THEN 1 ELSE 0 END) as pct_missing_phone,
      AVG(CASE WHEN ${leads.ownerEmail} IS NULL THEN 1 ELSE 0 END) as pct_missing_email,
      AVG(CASE WHEN ${leads.yearBuilt} < 2000 OR ${leads.yearBuilt} IS NULL THEN 1 ELSE 0 END) as pct_old_roof,
      AVG(${leads.leadScore}) as avg_score,
      AVG(CASE WHEN ${leads.ownerPhone} IS NOT NULL AND ${leads.ownerEmail} IS NOT NULL THEN 1 ELSE 0 END) as contactable_pct,
      AVG(${leads.latitude}) as center_lat,
      AVG(${leads.longitude}) as center_lng,
      MIN(${leads.latitude}) as min_lat, MAX(${leads.latitude}) as max_lat,
      MIN(${leads.longitude}) as min_lng, MAX(${leads.longitude}) as max_lng
    FROM ${leads}
    WHERE ${leads.zipCode} = ${zipCode} AND ${leads.marketId} = ${marketId}
  `);
  const s = stats.rows[0];
  if (!s.lead_count) return null;

  const stormRisk = Math.min(100, (s.avg_hail || 0) * 6);
  const roofAgeRisk = (s.pct_old_roof || 0) * 100;
  const dataGap = ((s.pct_missing_phone || 0) + (s.pct_missing_email || 0)) * 50;
  const valueScore = Math.min(100, (s.median_value || 0) / 150000);
  const densityScore = Math.min(100, s.lead_count / 80);

  const zipScore = Math.round(
    0.30 * stormRisk +
    0.20 * roofAgeRisk +
    0.20 * dataGap +
    0.15 * valueScore +
    0.15 * densityScore
  );

  const tile = {
    marketId,
    zipCode,
    zipScore,
    stormRiskScore: stormRisk,
    roofAgeScore: roofAgeRisk,
    dataGapScore: dataGap,
    propertyValueScore: valueScore,
    leadDensityScore: densityScore,
    contactabilityScore: (s.contactable_pct || 0) * 100,
    leadCount: s.lead_count,
    avgLeadScore: s.avg_score,
    avgHailEvents: s.avg_hail,
    medianPropertyValue: Math.round(s.median_value),
    pctMissingPhone: s.pct_missing_phone,
    pctMissingEmail: s.pct_missing_email,
    pctOldRoof: s.pct_old_roof,
    recommendedSpend: zipScore > 70 ? 1800 : zipScore > 50 ? 900 : 300,
    projectedEv: zipScore * 120, // rough
    centerLat: s.center_lat,
    centerLng: s.center_lng,
    boundingBox: {
      minLat: s.min_lat, maxLat: s.max_lat,
      minLng: s.min_lng, maxLng: s.max_lng,
    },
    lastComputedAt: new Date(),
  };

  await db.insert(zipTiles).values(tile).onConflictDoUpdate({
    target: [zipTiles.zipCode],
    set: tile,
  });

  return tile;
}

export async function scoreAllZips(marketId: string = 'dfw') {
  const zips = await db.execute(sql`SELECT DISTINCT ${leads.zipCode} FROM ${leads} WHERE ${leads.marketId} = ${marketId}`);
  const tiles = await Promise.all(zips.rows.map(z => computeZipTile(z.zip_code, marketId)));
  return tiles.filter(Boolean).sort((a, b) => b.zipScore - a.zipScore);
}

export async function getZipTiles(marketId: string) {
  return db.select().from(zipTiles).where(eq(zipTiles.marketId, marketId));
}
